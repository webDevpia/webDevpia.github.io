### 특성 엔지니어링 — 새로운 특성 만들기

> "기존 데이터를 조합해서 더 유용한 정보를 추출하는 것이 특성 엔지니어링입니다. 요리에서 재료를 다듬는 것과 같습니다."

```python
# 1) FamilySize: 혼자가 아닌 가족의 크기
df_work['family_size'] = df_work['sibsp'] + df_work['parch'] + 1
print("FamilySize 분포:")
print(df_work['family_size'].value_counts().sort_index())

# 2) IsAlone: 혼자 탑승했는지 여부
df_work['is_alone'] = (df_work['family_size'] == 1).astype(int)
print(f"\n혼자 탑승 비율: {df_work['is_alone'].mean():.1%}")

# 3) 가족 크기를 카테고리로
def family_category(size):
    if size == 1:
        return 'alone'
    elif size <= 4:
        return 'small'
    else:
        return 'large'

df_work['family_cat'] = df_work['family_size'].apply(family_category)
print("\n가족 규모 카테고리:")
print(df_work.groupby('family_cat')['survived'].agg(['count', 'mean']).round(3))
```

### 6가지 모델 비교

```python
# Pipeline으로 스케일링 + 모델을 묶어서 처리
models = {
    'Logistic Regression':     Pipeline([('scaler', StandardScaler()),
                                          ('model', LogisticRegression(max_iter=1000, random_state=42))]),
    'Decision Tree':           Pipeline([('scaler', StandardScaler()),
                                          ('model', DecisionTreeClassifier(random_state=42))]),
    'Random Forest':           Pipeline([('scaler', StandardScaler()),
                                          ('model', RandomForestClassifier(n_estimators=100, random_state=42))]),
    'K-Nearest Neighbors':     Pipeline([('scaler', StandardScaler()),
                                          ('model', KNeighborsClassifier())]),
    'SVM':                     Pipeline([('scaler', StandardScaler()),
                                          ('model', SVC(random_state=42, probability=True))]),
    'Gradient Boosting':       Pipeline([('scaler', StandardScaler()),
                                          ('model', GradientBoostingClassifier(random_state=42))]),
}

# 5-Fold 교차 검증으로 공정하게 비교
print("=" * 55)
print(f"{'모델':25s} | {'평균 정확도':10s} | {'표준편차':8s}")

results = {}
for name, pipeline in models.items():

print("=" * 55)
```
